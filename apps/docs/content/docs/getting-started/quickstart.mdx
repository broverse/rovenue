### Configure once at startup

<Tabs groupId="platform" persist items={['React Native', 'Swift', 'Kotlin']}>
  <Tab value="React Native">
    ```ts

    Rovenue.configure({ apiKey: 'rov_pub_...', baseUrl: 'https://edge.rovenue.app' });
    ```
  </Tab>
  <Tab value="Swift">
    ```swift

    Rovenue.configure(apiKey: "rov_pub_...", baseUrl: "https://edge.rovenue.app")
    ```
  </Tab>
  <Tab value="Kotlin">
    ```kotlin
    import dev.rovenue.sdk.Rovenue

    Rovenue.configure(apiKey = "rov_pub_...", baseUrl = "https://edge.rovenue.app")
    ```
  </Tab>
</Tabs>

<Step>
### Check an entitlement

<Tabs groupId="platform" persist items={['React Native', 'Swift', 'Kotlin']}>
  <Tab value="React Native">
    ```tsx

    const pro = useEntitlement('pro'); // reactive, cache-first
    ```
  </Tab>
  <Tab value="Swift">


    ```kotlin

    ```
  </Tab>
</Tabs>

<Step>
### Fetch offerings and make a purchase

The SDK fetches products directly from the store, drives the native purchase
sheet (StoreKit 2 on iOS, Play Billing on Android), and validates the receipt
server-side. Your app receives updated entitlements in the result.

<Tabs groupId="platform" persist items={['React Native', 'Swift', 'Kotlin']}>
  <Tab value="React Native">
    ```ts
    import {
      Rovenue,
      PurchaseCancelledError,
      PurchasePendingError,
    } from '@rovenue/react-native-sdk';

    const offerings = await Rovenue.getOfferings();
    const pkg = offerings.current?.packages[0];

    if (pkg) {
      try {
        const result = await Rovenue.purchase(pkg);
        // result.entitlements / result.creditBalance are already up to date
      } catch (e) {
        if (e instanceof PurchaseCancelledError) { /* user dismissed — ignore */ }
        else if (e instanceof PurchasePendingError) { /* Ask-to-Buy / deferred */ }
        else throw e;
      }
    }

    await Rovenue.restorePurchases();
    ```
  </Tab>
  <Tab value="Swift">
    ```swift
    let offerings = try await Rovenue.shared.getOfferings()
    if let pkg = offerings.current?.packages.first {
      do {
        let result = try await Rovenue.shared.purchase(pkg)
        // result.entitlements / result.creditBalance are already up to date
      } catch is PurchaseCancelledError { /* user dismissed — ignore */ }
        catch is PurchasePendingError   { /* Ask-to-Buy / deferred */ }
    }

    try await Rovenue.shared.restorePurchases()
    ```
  </Tab>
  <Tab value="Kotlin">
    ```kotlin
    val offerings = Rovenue.shared.getOfferings()
    val pkg = offerings.current?.packages?.firstOrNull()

    if (pkg != null) {
      try {
        val result = Rovenue.shared.purchase(pkg)
        // result.entitlements / result.creditBalance are already up to date
      } catch (e: PurchaseCancelledError) { /* user dismissed — ignore */ }
        catch (e: PurchasePendingError)   { /* Ask-to-Buy / deferred */ }
    }

    Rovenue.shared.restorePurchases()
    ```
  </Tab>
</Tabs>
</Step>
</Steps>

<Callout type="info">
On purchase success, the SDK refreshes entitlements and credits automatically and emits
a change event. See [Reacting to changes](/docs/guides/reacting-to-changes).
</Callout>
